Создать два сложных запроса с использованием объединения JOIN и без использования
подзапросов.

1.
CREATE TEMPORARY TABLE big_video(
    id INT,
    url VARCHAR(250),
    size INT,
    owner_id INT
);

INSERT INTO big_video
    SELECT id, url, size, owner_id
        FROM video
    ORDER BY size DESC
    LIMIT 10;

SELECT big_video.id,
    first_name,
    last_name,
    photo.url AS main_photo_url,
    big_video.url AS video_url,
    big_video.size
FROM users
    JOIN big_video
        ON big_video.owner_id = users.id
    LEFT JOIN photo
        ON photo.id = users.main_photo_id;

2.
WITH big_video AS
    (SELECT id, url, size, owner_id FROM video
    ORDER BY size DESC LIMIT 10)
SELECT big_video.id,
    first_name,
    last_name,
    photo.url AS main_photo_url,
    video.url AS video_url,
    video.size
FROM users
    JOIN big_video
        ON big_video.owner_id = users.id
    LEFT JOIN photo
        ON photo.id = users.main_photo_id
ORDER BY size DESC;
